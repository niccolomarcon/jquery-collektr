 $(document).ready(function(){
  $("#placeholder").collektr({
   token: "API_token",
   selector: "my-board",
   columnNumber: 2
   templateURL: "customTemplate.html",
   customHelpers: function(){
    //override the parseDate helper
    Handlebars.registerHelper('parseDate', function(_date, options) {
     months = [ "Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Oct", "Nov", "Dec" ];
     date = new Date(_date)
     return date.getDate() + " " + months[date.getMonth()] + " " + date.getFullYear();
    });
   }
  });
 });
